  <script>
    function showPage(id) {
      document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
      document.getElementById(id).classList.add('active');
    }

    function turnOn() {
      document.getElementById("lamp").classList.add("on");
      document.getElementById("wireLeft").style.display = "block";
      document.getElementById("wireRight").style.display = "block";
    }

    function turnOff() {
      document.getElementById("lamp").classList.remove("on");
      document.getElementById("wireLeft").style.display = "none";
      document.getElementById("wireRight").style.display = "none";
    }

    function startVolcano() {
      const eruption = document.getElementById("eruption");
      eruption.style.display = "block";
      eruption.style.animation = "none";
      void eruption.offsetWidth;
      eruption.style.animation = "erupt 1s ease-out forwards";

      const dropsContainer = document.getElementById("drops");
      dropsContainer.innerHTML = "";
      for (let i = 0; i < 6; i++) {
        const drop = document.createElement("div");
        drop.className = "magma-drop";
        drop.style.left = `${65 + (Math.random() * 40 - 20)}px`;
        drop.style.animationDelay = `${i * 0.3}s`;
        dropsContainer.appendChild(drop);
      }

        "NaHCO‚ÇÉ + CH‚ÇÉCOOH ‚Üí CO‚ÇÇ + H‚ÇÇO + CH‚ÇÉCOONa";
